function userFieldsFilter(model, data, error) {
  if (!error && data && model.modelName === 'user') {
    if (data.salt) {
      data.salt = undefined      
      delete data.salt
    }
    if (data.hashed_password) {
      data.hashed_password = undefined
      delete data.hashed_password
    }
  }
  return { data, error }
}
